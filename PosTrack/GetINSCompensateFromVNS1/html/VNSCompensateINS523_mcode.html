123%% 位移补偿
124%%% Input
125% trackedMakerPosition_InertialTime ： 马克点光学位置，按惯性时序存储
126% HipDisplacement ： Hip在北东地下的位置 （由 BVH 得到）
127% InertialPosition： 惯性系与马克点安装位置对应关节的位置（安装在头上时，InertialPosition 为惯性头的位置）
128%%% Output
129% InertialPositionNew ： 补偿后惯性的位置
130% HipDisplacementNew： 补偿后Hip的位置
131function [ InertialPositionCompensate_out,HipDisplacementNew_out ] = VNSCompensateINS...
132( compensateRate,trackedMakerPosition_InertialTime,HipDisplacement,InertialPosition )
133coder.inline('never');
134coder.extrinsic('fprintf');
135coder.extrinsic('DrawCompensate');
136global CalStartIN  CalEndIN 
137
138%% BVH 读取的个数 N_BVH 可能会比 N1 多几个
139N_BVH = size(HipDisplacement,2);
140N1 = size(trackedMakerPosition_InertialTime,2);
141for k=N1+1:N_BVH
142    trackedMakerPosition_InertialTime(:,k) = trackedMakerPosition_InertialTime(:,N1);
143    InertialPosition(:,k) = InertialPosition(:,N1);
144end
145
146%% 先补偿 InertialPositionNew
147persistent InertialPositionCompensate  InertialPositionNew HipDisplacementNew
148if isempty(InertialPositionCompensate)
149    InertialPositionCompensate = NaN(3,N_BVH); % 每一步的累积位移补偿量 记录
150    InertialPositionCompensate(:,1)=zeros(3,1);
151    InertialPositionNew = NaN(3,N_BVH);
152    InertialPositionNew(3,:) = InertialPosition(3,:); % 高度不补偿
153    InertialPositionNew(1:2,1) = InertialPosition(1:2,1) ;  % 起始点选择惯性
154    
155    HipDisplacementNew = NaN(3,N_BVH); 
156    HipDisplacementNew(3,:) = HipDisplacement(3,:);
157    HipDisplacementNew(:,1) = HipDisplacement(:,1);
158end
159persistent  InertialErr StepCompensate
160if isempty(StepCompensate)
161    InertialErr = zeros(2,N_BVH);       % 实时记录补偿后的误差
162    StepCompensate = zeros(2,N_BVH);  % 单步补偿量
163end
164
165for k = max(2,CalStartIN) : CalEndIN 
166    % 先用纯惯性递推
167    InertialPositionNew(1:2,k) = InertialPositionNew(1:2,k-1)+( InertialPosition(1:2,k)-InertialPosition(1:2,k-1) );
168    % 先求纯惯性为误差
169    if ~isnan(trackedMakerPosition_InertialTime(1,k))
170        InertialErr(:,k) = trackedMakerPosition_InertialTime(1:2,k) - InertialPositionNew(1:2,k) ;
171        % 补偿误差
172        InertialPositionNew(1:2,k) = InertialPositionNew(1:2,k) + InertialErr(:,k)*compensateRate ; 
173        StepCompensate(:,k) = InertialErr(:,k)*compensateRate ;
174    end
175    InertialPositionCompensate(1:2,k) = InertialPositionNew(1:2,k) - InertialPosition(1:2,k) ;  % 累积位移补偿量
176end
177
178%% 通过 InertialPositionNew 计算 HipDisplacementNew
179% 将Head位置传到Hip，：保持头和head的相对位移
180
181for k=CalStartIN:CalEndIN    
182    HipDisplacementNew( 1:2,k ) = HipDisplacement(1:2,k) + InertialPositionCompensate(1:2,k) ;
183end
184
185InertialPositionCompensate_out = InertialPositionCompensate ;
186HipDisplacementNew_out = HipDisplacementNew;
187if CalEndIN >= size(trackedMakerPosition_InertialTime,2)-10
188    DrawCompensate( InertialErr,StepCompensate,trackedMakerPosition_InertialTime,...
189        InertialPosition,InertialPositionNew,InertialPositionCompensate,compensateRate,N_BVH ) ;
190end
