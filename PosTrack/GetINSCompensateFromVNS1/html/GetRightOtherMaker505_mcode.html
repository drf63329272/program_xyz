  9%% 判断思路：比较2个相对位移矢量：1）dT(3 sec)运动时间时  2）dS（1m）运动位移长度时
 10% 1)dT(3 sec)时间内，惯性和视觉位移向量的大小差<0.1m，方向差<60°（当位移矢量长度小于0.2m时不比较方向）

 12function [ trackedMakerPosition_InertialTime_Out,otherMakers ] = ...
 13    GetRightOtherMaker( otherMakers,InertialPosition )
 14coder.extrinsic('fprintf');
 15coder.extrinsic('DrawTracedINSVNS');
 16
 17global    visionFre inertialFre 
 18global      IsGetFirstMarker 
 19global      CalStartVN CalEndVN  CalStartIN  CalEndIN  
 20
 21%% 阈值参数设置
 22persistent makerTrackThreshold INSVNSCalibSet 
 23if isempty(makerTrackThreshold)
 24    [ makerTrackThreshold,INSVNSCalibSet ] = SetParameters( );
 25end
 26
 27IsGetFirstMarker = 0 ;
 28
 29dT_Ninertial = fix(makerTrackThreshold.moveTime*inertialFre) ;
 30
 31%% 在线/离线 第一次要求预定义变量大小
 32persistent   trackedMakerPosition_InertialTime  trackedMakerPosition TrackFlag trackedMarkerVelocity  InertialVelocity
 33persistent INSVNSCalib_VS_k  Calib_N_New  IsCalibDataEnough
 34MarkerTN = length(otherMakers);   %  视觉时间长度
 35InertialN = size(InertialPosition,2);
 36
 37if isempty(InertialVelocity)
 38    TrackFlag = zeros(1,MarkerTN);
 39    trackedMakerPosition = NaN(3,MarkerTN); % 判断成功的马克点位置
 40    trackedMakerPosition_InertialTime = NaN( 3,InertialN );
 41    
 42    InertialVelocity = NaN(5,InertialN);
 43    trackedMarkerVelocity = NaN(5,MarkerTN); 
 44    
 45    INSVNSCalib_VS_k = NaN(2,50); % INSVNSCalib_VS_k（:,i）存储一段位移
 46    Calib_N_New = 0;   % 搜索乘的位移段个数
 47    IsCalibDataEnough = 0;
 48end
 49%% 这些判断指标只在离线时记录
 50persistent dPi_ConJudge dPError_dT_xy  dPError_dT_z  dPError_dS_xyNorm dP_Inertial_xyNorm  angleErr_dS  angleErr_dT_Min ConTrackedFailed INSVNSMarkHC_Min
 51if isempty(dPi_ConJudge)
 52    dPi_ConJudge = NaN(1,MarkerTN);   
 53    dPError_dT_xy = NaN(1,MarkerTN);  
 54    dPError_dT_z = NaN(1,MarkerTN);  
 55    dPError_dS_xyNorm = NaN(1,MarkerTN);  
 56    dP_Inertial_xyNorm = NaN(1,MarkerTN);  
 57    angleErr_dS = NaN(1,MarkerTN);  
 58    angleErr_dT_Min = NaN(1,MarkerTN); 
 59    ConTrackedFailed = zeros(1,MarkerTN);
 60    INSVNSMarkHC_Min = NaN(1,MarkerTN);               
 61end
 62persistent Crw
 63if isempty(Crw)
 64   Crw = eye(3); 
 65end
 66%% 求惯性速度
 67for k=CalStartIN:CalEndIN
 68    [ InertialVelocity_i,k_calV ] = CalVelocity( InertialPosition,k,inertialFre,INSVNSCalibSet.dT_CalV_Calib,INSVNSCalibSet.MinXYVNorm_CalAngle );
 69    if k_calV>0 
 70        InertialVelocity(:,k_calV) = InertialVelocity_i ;
 71    end
 72end
 73%% 单马克点跟踪 
 74coder.extrinsic('waitbar');
 75coder.extrinsic('close');
 76%  wh = waitbar(0,'SearchDistanceK');
 77for k=CalStartVN:CalEndVN
 78    otherMakers_k = otherMakers(k) ;  
 79    inertial_k = otherMakers_k.inertial_k ;
 80        
 81    %  last_dT_k
 82    inertial_dT_k_last = inertial_k - dT_Ninertial ;
 83    inertial_dT_k_last = max(inertial_dT_k_last,1);        
 84        
 85    if k>1
 86        otherMakers_k_last = otherMakers(k-1) ; 
 87        if coder.target('MATLAB') && isnan(otherMakers_k_last.ContinuesFlag)
 88           fprintf('判定第%0.0f个点时，前一个otherMakers点没有结果\n',k); 
 89        end
 90        % otherMakers_k_last 必须是更新了判定结果的，检查一下
 91        
 92        
 93%         if k>2 && isnan(trackedMakerPosition(1,k-2))
 94%             if sum(otherMakers_k_last.ContinuesFlag==1)~=0
 95%                disp('err') 
 96%             end
 97%         end
 98    else
 99        otherMakers_k_last = GetEmpty_otherMakers(otherMakers_k) ;
100    end
101
102    [ trackedMakerPosition(:,k),otherMakersNew_k,TrackFlag(k),JudgeIndex  ] = JudgeMaker...
103        ( otherMakers_k,otherMakers_k_last,k,inertial_k,trackedMakerPosition,InertialPosition,inertial_dT_k_last,...
104        makerTrackThreshold ) ;
105        
106    if coder.target('MATLAB')
107        dPi_ConJudge(k) = JudgeIndex.dPi_ConJudge  ;
108        dPError_dT_xy(k) = JudgeIndex.dPError_dT_xy ;
109        dPError_dT_z(k) = JudgeIndex.dPError_dT_z ;
110        dPError_dS_xyNorm(k) = JudgeIndex.dPError_dS_xyNorm ;
111        dP_Inertial_xyNorm(k) = JudgeIndex.dP_Inertial_xyNorm ;
112        angleErr_dT_Min(k) = JudgeIndex.angleErr_dT_Min ;
113        angleErr_dS(k) = JudgeIndex.angleErr_dS ;
114        INSVNSMarkHC_Min(k) = JudgeIndex.INSVNSMarkHC_Min ;
115    end
116       
117    if sum(otherMakersNew_k.ContinuesFlag==1)~=0 && isnan(trackedMakerPosition(1,k))
118        ConTrackedFailed(k) = 1 ;
119    else
120        ConTrackedFailed(k) = 0 ;
121    end
122    
123    %% 求  INSMarkH0  VNSMarkH0
124    if isnan(makerTrackThreshold.INSMarkH0) && ~isnan(trackedMakerPosition(1,k))
125        fprintf( '第一个点搜素成功 time = %0.1f sec, TrackFlag(%d) = %0.1f ,',k/visionFre,k,TrackFlag(k) );
126        % 检测Hip的俯仰和横滚小才取值！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！
127        makerTrackThreshold.INSMarkH0 = - InertialPosition(3,inertial_k);
128        makerTrackThreshold.VNSMarkH0 = - trackedMakerPosition(3,k) ;
129        
130 
131%         HipQuaternion_k = HipQuaternion( :,inertial_k );
132%         HipAttitude = GetHipAttitude( HipQuaternion_k );
133%         
134%         HipQuaternion_k = Qinv( HipQuaternion_k ) ; % 四元数定义由 顺时针 改为 逆时针。
135%         
136%         CHip_k = Q2C(HipQuaternion_k);
137%         
138%         C_HipLUF_NED0 = RotateX(pi/2) * RotateY(-pi/2);  % Hip 的左上前系 到 NED的0姿态系
139%         C_NED_HipNED0 = C_HipLUF_NED0 * CHip_k ;               
140%         Attitude = C2Euler( C_NED_HipNED0,'ZYX' )*180/pi
141        
142        
143        %% 第一次跟踪马克点成功
144        % 标定视觉与光学的原点（不考虑方向差）
145%         Xrw_r = trackedMakerPosition(1:2,k) - InertialPosition(1:2,inertial_k);
146%         Xrw_r = [Xrw_r;0];
147%         N_otherMakers = length( otherMakers );
148%         for i=1:N_otherMakers
149%             if ~isempty(otherMakers(i).Position)
150%                 m = size( otherMakers(i).Position,2 );
151%                 otherMakers(i).Position = otherMakers(i).Position - repmat(Xrw_r,1,m) ;
152%                 trackedMakerPosition(:,i) = trackedMakerPosition(:,i) - Xrw_r ;
153%             end
154%         end
155    end
156    %% 求视觉速度
157    [ trackedMarkerVelocity_k,k_calV ] = CalVelocity...
158        ( trackedMakerPosition,k,visionFre,INSVNSCalibSet.dT_CalV_Calib,INSVNSCalibSet.MinXYVNorm_CalAngle );
159    if k_calV>0 
160        trackedMarkerVelocity(:,k_calV) = trackedMarkerVelocity_k ;
161        if IsCalibDataEnough==0   % 只搜索一次
162        %% 搜索可用于标定的数据
163    %         dbstop in SearchCalibData
164            Calib_N_Last = Calib_N_New;
165            [ INSVNSCalib_VS_k,Calib_N_New,IsCalibDataEnough,dX_Vision ] = SearchCalibData...
166                ( INSVNSCalib_VS_k,Calib_N_Last,trackedMarkerVelocity,trackedMakerPosition,k_calV,INSVNSCalibSet ) ;
167            if IsCalibDataEnough==1
168                fprintf( '第一个用于坐标系映射计算的点OK，time = %0.1f sec \n',k/visionFre );
169                Crw = INSVNSCalib( INSVNSCalib_VS_k,Calib_N_New,dX_Vision,InertialPosition );                 
170            end
171        end        
172    end
173    otherMakersNew_k = CompensateSecond( otherMakersNew_k,Crw  );
174    otherMakers(k)=otherMakersNew_k;     % 更新 otherMakers(k)
175    %% 转成惯性马克点的时序
176    
177    trackedMakerPosition_InertialTime(:,inertial_k) = trackedMakerPosition(:,k) ;
178%     
179%     if mod(k,fix(MarkerTN/10))==0
180%         waitbar(k/MarkerTN);
181%     end
182end
183%  close(wh);
184 %% Output
185trackedMakerPosition_InertialTime_Out = trackedMakerPosition_InertialTime ;
186 
187 %% result analyse
188
189if coder.target('MATLAB') && CalEndVN == MarkerTN
190     trackedMaker.trackedMakerPosition = trackedMakerPosition ;
191%      trackedMaker.time = otherMakersTime ;
192     trackedMaker.MarkerSet = otherMakers(1).MarkerSet ;
193     N = length(trackedMakerPosition) ;
194    for k=1:N
195        otherMakers(k).trackedMakerPosition = trackedMakerPosition(:,k) ;
196    end
197    for k=1:N
198        i = N-k+1;
199        if ~isnan(trackedMakerPosition(1,i))
200            trackedMaker.LastMakerPosition = trackedMakerPosition(:,i);
201            trackedMaker.LastMakerTime = otherMakers(i).time ;
202            break;
203        end
204    end
205    
206    DrawTracedINSVNS( TrackFlag,trackedMaker,InertialPosition,trackedMarkerVelocity,InertialVelocity,INSVNSCalib_VS_k,INSVNSCalibSet ) ;
207    
208end

211    otherMakersTime = ( 1:MarkerTN )/1;

216    % xlabel('time sec')

219    plot(otherMakersTime,dPi_ConJudge)
220    temp = makerTrackThreshold.MaxContinuesDisplacement ;
221    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
222
223    xlabel('time sec')
224
225    figure('name','INSVNSMarkHC_Min')
226    plot(otherMakersTime,INSVNSMarkHC_Min)
227    temp = makerTrackThreshold.MaxMarkHighChange ;
228    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
229    temp = -makerTrackThreshold.MaxMarkHighChange ;
230    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
231    xlabel('time sec')
232
233
234    figure('name','dPError_dT_xy')
235    plot(otherMakersTime,dPError_dT_xy)
236    temp = makerTrackThreshold.PositionErrorBear_dT ;
237    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
238    xlabel('time sec')
239    legend( 'dPError\_dT\_xy','PositionErrorBear\_dT' )
240
241    figure('name','dPError_dT_z')
242    plot(otherMakersTime,dPError_dT_z)
243    temp = makerTrackThreshold.PositionErrorBear_dT ;
244    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
245    xlabel('time sec')
246    legend( 'dPError\_dT\_z','PositionErrorBear\_dT' )
247
248    figure('name','dP_Inertial_xyNorm')
249    plot(otherMakersTime,dP_Inertial_xyNorm)

251    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
252    xlabel('time sec')
253    legend( 'dPError\_dT','MaxStaticDisp\_dT' )

255    figure('name','angleErr_dT_Min')
256    plot(otherMakersTime,angleErr_dT_Min*180/pi)
257    temp = makerTrackThreshold.Max_dPAngle_dS*180/pi ;
258    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )
259    xlabel('time sec')
260    legend( 'dPError\_dT','MaxStaticDisp\_dT' )
261
262
263    figure('name','dPError_dS_xyNorm')
264    plot(otherMakersTime,dPError_dS_xyNorm)
265    temp = makerTrackThreshold.MaxPositionError_dS ;

267    xlabel('time sec')
268
269    figure('name','angleErr_dS')

272    line( [otherMakersTime(1) otherMakersTime(MarkerTN)],[temp temp],'color','r' )

276    plot(ConTrackedFailed)
