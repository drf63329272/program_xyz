1%% xyz 2015.6.1
 2
 3%% 计算某个点的速度
 4%% 得到 第 k 个点的位置后，计算的是第 k_calV 个点的速度（ k_calV = data_k-dN_CalV ;）
 5% dT_CalV： 速度计算步长
 6%  V(k) = (X(k+dN_CalV) - X(k-dN_CalV)) / (dT_CalV*2)
 7
 8% fre： 频率
 9% dT_CalV： 速度计算的步长时间
10
11%% Velocity_k
12% Velocity_k(1:3,1) : xyz三维的速度
13% Velocity_k(4,1) : xy平面速度的模
14% Velocity_k(5,1) : xy平面速度与正前方之间的夹角（模大于0.2m/s时才计算）
15
16
17function [ Velocity_k,k_calV ] = CalVelocity( Position,data_k,fre,dT_CalV,MinXYVNorm_CalAngle )
18% 速度计算的步长个数
19dN_CalV = fix(dT_CalV*fre) ;
20dN_CalV = max(dN_CalV,2);
21dN_CalV = min(dN_CalV,7);
22
23angleXY = NaN ;
24Velocity_k = NaN(5,1) ;
25
26%% 计算速度： 
27% 计算第 k_calV 个数的速度：用 [ k_calV-dN_CalV, k_calV+dN_CalV  ] 的一段数据
28% 得到 data_k 数据后，计算第 data_k-dN_CalV 个数的速度
29k_calV = data_k-dN_CalV ;
30if k_calV-dN_CalV<1 || k_calV+dN_CalV>length(Position)    
31    return; 
32end
33
34Position1 = Position( :,k_calV+dN_CalV ) ;
35Position2 = Position( :,k_calV-dN_CalV ) ;
36if isnan( Position1(1) ) || isnan( Position2(1) )
37    Velocity_k = NaN(5,1) ;
38    return;  % 计算速度的数据中有跟踪失败的点 
39end
40% xyz三维的速度
41Velocity_k = ( Position1 - Position2 ) / (dT_CalV*2) ;
42
43% 第四行存储xy平面速度的模
44trackedMarkerVelocity_xyNorm = normest(Velocity_k(1:2)) ; 
45
46% 第五行存储xy平面速度与正前方之间的夹角（模大于0.2m/s时才计算）
47if trackedMarkerVelocity_xyNorm >  MinXYVNorm_CalAngle 
48   temp =  [0 1]*Velocity_k(1:2) / trackedMarkerVelocity_xyNorm ;
49   angleXY = acos(temp);
50   if Velocity_k(1)>0
51       angleXY = -angleXY ;
52   end
53end
54Velocity_k = [ Velocity_k; trackedMarkerVelocity_xyNorm;angleXY ];
