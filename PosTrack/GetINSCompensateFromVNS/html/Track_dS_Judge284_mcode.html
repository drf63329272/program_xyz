499%% 马克点判定 4) 寻找惯性运动 dS 长度(大于 moveDistance)，且马克点跟踪成功的是时刻，判定 dP_Inertial 和 dP_Vision：
500function [ trackedMakerPosition_k_OK,TrackFlag,dPError_dS_Norm,angleErr_dS ] = Track_dS_Judge...
501    ( InertialPosition,inertial_k,trackedMakerPosition,otherMakersPosition_k_min,TrackFlag,makerTrackThreshold )
502global InertialData_visual_k
503
504trackedMakerPosition_k_OK=  NaN ;
505%% dS 位移长度段的位移差：同时考虑位移差大小和方向
506% find the point which moved moveDistance
507moveDistance = makerTrackThreshold.moveDistance;
508dS_Inertial_last_k = SearchDistanceK( InertialPosition,inertial_k,moveDistance,trackedMakerPosition ) ;  
509if isnan(dS_Inertial_last_k)
510    % 找不到运动大于 dS 长度且trackedMakerPosition中跟踪到的点
511%     fprintf('找不到运动固定长度的点，不进行验证<2>，跟踪失败。\n');
512    dPError_dS_Norm = -0.2 ;    % 给负值表示没有找到
513    angleErr_dS = -10*pi/180;
514    TrackFlag = -1.4 ;
515    return; 
516end
517dS_Vision_last_k = InertialData_visual_k(dS_Inertial_last_k)  ;
518
519dP_Inertial = InertialPosition(:,inertial_k) - InertialPosition(:,dS_Inertial_last_k);
520dP_Vision = otherMakersPosition_k_min - trackedMakerPosition(:,dS_Vision_last_k);
521dPError_dS = dP_Inertial-dP_Vision ;
522dPError_dS_Norm = normest(dPError_dS) ;
523temp = dP_Inertial'*dP_Vision / normest(dP_Inertial) / normest(dP_Vision) ;
524angleErr_dS = acos(temp) ;
525
526if dPError_dS_Norm < makerTrackThreshold.MaxPositionError_dS  && angleErr_dS < makerTrackThreshold.Max_dPAngle_dS
527    % 距离和角度 满足
528    trackedMakerPosition_k_OK = otherMakersPosition_k_min ;
529    TrackFlag = 4+TrackFlag ;
530%     fprintf('3.1）3.2） 模=%0.3f，角度差=%0.3f，跟踪OK \n',normest(dPError_dS),angleErr_dS*180/pi);
531else
532    TrackFlag = -TrackFlag ;
533%     fprintf('3.1）3.2） 模=%0.3f，角度差=%0.3f，跟踪失败 \n',normest(dPError_dS),angleErr_dS*180/pi);
534end
