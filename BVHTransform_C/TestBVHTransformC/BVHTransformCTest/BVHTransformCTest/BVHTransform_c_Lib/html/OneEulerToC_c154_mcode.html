22function C =  OneEulerToC_c( euler,rotateOrder,rotateDirection )
23coder.inline('never');
24
25for i=1:3
26    euler(i) = euler(i)*rotateDirection(i) ;
27end
28
29switch rotateOrder
30    case 'ZYX' 
31        C = RotateX(euler(3)) * RotateY(euler(2)) * RotateZ(euler(1)) ;
32    case 'ZXY'
33        C = RotateY(euler(3)) * RotateX(euler(2)) * RotateZ(euler(1)) ;
34    case 'XYZ'
35        C = RotateZ(euler(3)) * RotateY(euler(2)) * RotateX(euler(1)) ;
36    case 'XZY'
37        C = RotateY(euler(3)) * RotateZ(euler(2)) * RotateX(euler(1)) ;
38    case 'YXZ'
39        C = RotateZ(euler(3)) * RotateX(euler(2)) * RotateY(euler(1)) ;
40    case 'YZX'
41        C = RotateX(euler(3)) * RotateZ(euler(2)) * RotateY(euler(1)) ;
42end
