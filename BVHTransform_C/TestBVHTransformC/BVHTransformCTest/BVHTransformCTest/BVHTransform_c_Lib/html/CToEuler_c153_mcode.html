 3% direction cosine matrix  to Euler angle  for multi C
 4% C: [N*3*3]
 5%       euler = [eu_x,eu_y,eu_z]  ( Nframes*3 )
 6% store order of euler angle: the rotate order
 7% positive:anticlockwise rotate 
 8% the first and third euler angles scope: [-pi,pi]
 9% the second euler angles scope: [-pi/2,pi/2]
10% unit:rad 
11% rotateDirection: anticlockwise(1) or clockwise(-1)  111
12
13function euler = CToEuler_c( C,rotateOrder,rotateDirection )
14
15Nframes = size( C,1 );
16euler = zeros( Nframes,3 );
17for k=1:Nframes
18    C_K = permute( C(k,:,:),[2,3,1] );
19    euler(k,:) = OneCToEuler_c( C_K,rotateOrder,rotateDirection );
20end
