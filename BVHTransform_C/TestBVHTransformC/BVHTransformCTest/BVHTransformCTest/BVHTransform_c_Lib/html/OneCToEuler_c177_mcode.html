23function euler = OneCToEuler_c( C,rotateOrder,rotateDirection )
24
25switch rotateOrder
26    case 'ZYX' 
27        euler1 = atan2( C(1,2),C(1,1) ) ;
28        euler2 = asin( -C(1,3) ) ;
29        % euler2 = atan2( -C(1,3),sqrt( C(3,2)^2+C(3,3)^2 ) ) ;
30        euler3 = atan2( C(2,3),C(3,3) ) ;
31    case 'ZXY'
32        euler1 = atan2( -C(2,1),C(2,2) ) ;
33        euler2 = asin( C(2,3) ) ;
34        % euler2 = atan2( C(2,3),sqrt( C(2,1)^2+C(2,2)^2 ) ) ;
35        euler3 = atan2( -C(1,3),C(3,3) ) ;
36    case 'XYZ'
37        euler1 = atan2( -C(3,2),C(3,3) ) ;
38        euler2 = asin(C(3,1)) ;
39        euler3 = atan2( -C(2,1),C(1,1) ) ;
40    case 'XZY'
41        euler1 =  atan2(C(2,3),C(2,2)) ;
42        euler2 =  asin(-C(2,1)) ;
43        euler3 =  atan2(C(3,1),C(1,1)) ;
44    case 'YXZ'
45        euler1 = atan2( C(3,1),C(3,3) ) ;
46        euler2 = asin( -C(3,2) ) ;
47        % euler2 = atan2( -C(3,2),sqrt( C(1,2)^2+C(2,2)^2 ) ) ;
48        euler3 = atan2( C(1,2),C(2,2) ) ;        
49    case 'YZX'  
50        euler1 = atan2(-C(1,3),C(1,1))  ;
51        euler2 = asin(C(1,2)) ;
52        euler3 = atan2(-C(3,2),C(2,2)) ;
53end
54euler = [ euler1; euler2; euler3 ];
55
56for i=1:3
57    euler(i) = euler(i)*rotateDirection(i) ;
58end
