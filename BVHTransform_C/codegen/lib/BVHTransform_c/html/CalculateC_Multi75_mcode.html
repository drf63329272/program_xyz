259function C13 = CalculateC_Multi( C12,C23 )
260coder.inline('never');
261
262C13 = coder.nullcopy(zeros(size(C12)));
263
264if size(C12,3)==1
265    C13 = C23*C12 ;
266    return;
267end
268
269Nframes = size( C12,1 );
270C13 = zeros(Nframes,3,3);
271for k=1:Nframes
272    C23_k = permute( C23(k,:,:),[2,3,1] );
273    C12_k = permute( C12(k,:,:),[2,3,1] );
274    C13(k,:,:) = C23_k*C12_k ;
275end
