 83%% 判断搜索得到的位移数据够不够多，是否已经均匀分布
 84function [ IsCalibDataVelocityOK,dX_Vision ] = JudgeIsCalibDataEnough( INSVNSCalib_VS_k,Calib_N_New,trackedMakerPosition,INSVNSCalibSet )
 85
 86angleUniformityErr = INSVNSCalibSet.angleUniformityErr ;
 87
 88M = Calib_N_New;
 89dX_Vision = zeros(3,M);
 90dX_Angle = zeros(1,M);
 91HaveBiggerData = 0;
 92HaveSmallerData = 0;
 93IsCalibDataVelocityOK = 0;
 94for k=1:M
 95    dX_Vision(:,k) = trackedMakerPosition(:,INSVNSCalib_VS_k(2,k)) - trackedMakerPosition(:,INSVNSCalib_VS_k(1,k))  ;  % 从 开始 指向 结束
 96    %% 只考虑平面内的位移
 97    dX_Vision(3,k) = 0; 
 98    %% 根据所有位移矢量与第一个矢量的夹角判断是否分布均匀
 99    angle = acos( dX_Vision(:,1)'*dX_Vision(:,k)/normest(dX_Vision(:,1))/normest(dX_Vision(:,k)) );
100    % 通过叉乘可判断角度方向
101    temp = cross(dX_Vision(:,1),dX_Vision(:,k)) ;
102    if temp(3)<0
103        % 从 dX_Vision(:,1) 到 dX_Vision(:,k) 逆时针转动超过180°
104        angle = -angle ;
105    end
106    
107    % 以dX_Vision(:,1)为中心，对所有的位移矢量，若与dX_Vision(:,1)夹角大于90°则反号
108    if angle > pi/2
109        angle = angle-pi ;
110    end
111    if angle < -pi/2
112        angle = angle+pi ;
113    end    
114    
115    dX_Angle(k) = angle ;
116    % 在 [60-angleUniformityErr,60+angleUniformityErr] 和
117    % [-60-angleUniformityErr,-60+angleUniformityErr] 范围内均存在位移矢量时判定分布均匀
118    if dX_Angle(k) > pi/3-angleUniformityErr && dX_Angle(k) < pi/3+angleUniformityErr
119       HaveBiggerData = 1 ; 
120    end
121    if dX_Angle(k) > -pi/3-angleUniformityErr && dX_Angle(k) < -pi/3+angleUniformityErr
122       HaveSmallerData = 1 ; 
123    end
124    if HaveSmallerData==1 && HaveBiggerData==1
125       %% 判定位移矢量满足分布均匀条件
126       IsCalibDataVelocityOK = 1 ;
127    end
128end
129%% 均匀性条件满足时将所有的位移矢量绘制出来
130if IsCalibDataVelocityOK==1
131   figure('name','dX_Vision') 
132   hold on
133   for k=1:M
134       if k==1
135            plot( [-dX_Vision(1,k) dX_Vision(1,k)],[-dX_Vision(2,k) dX_Vision(2,k)],'r' ); 
136            hold on 
137            plot(dX_Vision(1,k),dX_Vision(2,k),'.r')
138       else
139           plot( [-dX_Vision(1,k) dX_Vision(1,k)],[-dX_Vision(2,k) dX_Vision(2,k)] ); 
140           hold on 
141           plot(dX_Vision(1,k),dX_Vision(2,k),'.k')
142       end
143   end
144   dX_Angle_Degree = dX_Angle*180/pi;
145end
