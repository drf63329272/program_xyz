 11% dataFlag： dataWrite 的数据类型
 12function otherMakersContinues = Write_otherMakersContinues_i...
 13    ( otherMakersContinues,i_marker,dataWrite,dataFlag,dataN_i_j )
 14
 15switch dataFlag
 16    case 0  % dataWrite  为 data_i
 17        data_i  = dataWrite;
 18    case {1,2,3,4} % dataWrite 为 ConPosition_i   
 19        
 20        data_i = Read_otherMakersContinues_i( otherMakersContinues,i_marker );        
 21        % 更新数据
 22        switch dataFlag
 23            case 1
 24                M1 = 1;        % ConPosition_i
 25                M2 = 3; 
 26                %  更新 dataN
 27                otherMakersContinues.dataN(M1:M2,i_marker) = repmat(dataN_i_j,3,1) ;
 28            case 2
 29                M1 = 4;        % ConVelocity_i
 30                M2 = 8; 
 31                %  更新 dataN
 32                otherMakersContinues.dataN(M1:M2,i_marker) = repmat(dataN_i_j,5,1) ;
 33            case 3
 34                M1 = 9;        % ConAcc_i
 35                M2 = 13; 
 36                %  更新 dataN
 37                otherMakersContinues.dataN(M1:M2,i_marker) = repmat(dataN_i_j,5,1) ;
 38            case 4
 39                M1 = 14;        % ConAccWaveFlag_i
 40                M2 = 27; 
 41                %  更新 dataN
 42                otherMakersContinues.dataN(14:16,i_marker) = dataN_i_j ;
 43                otherMakersContinues.dataN(17:27,i_marker) = repmat(max(dataN_i_j),11,1)  ;
 44                
 45            otherwise
 46                disp('error-1 in Write_otherMakersContinues_i');
 47                otherMakersContinues = NaN;
 48                return;
 49        end
 50        
 51        %  更新 data_i
 52        data_i( M1:M2,1:size(dataWrite,2) ) = dataWrite;  
 53    otherwise
 54        disp('error-2 in Write_otherMakersContinues_i');
 55        otherMakersContinues = NaN;
 56        return;
 57end
 58
 59switch i_marker
 60    case 1
 61        otherMakersContinues.data1 = data_i;
 62    case 2
 63        otherMakersContinues.data2 = data_i;
 64    case 3
 65        otherMakersContinues.data3 = data_i;
 66    case 4
 67        otherMakersContinues.data4 = data_i;
 68    case 5

 70    case 6
 71        otherMakersContinues.data6 = data_i;
 72    case 7
 73        otherMakersContinues.data7 = data_i;
 74    case 8
 75        otherMakersContinues.data8 = data_i;
 76    case 9
 77        otherMakersContinues.data9 = data_i;
 78    case 10
 79        otherMakersContinues.data10 = data_i;
 80    case 11
 81        otherMakersContinues.data11 = data_i;
 82    case 12
 83        otherMakersContinues.data12 = data_i;
 84    case 13
 85        otherMakersContinues.data13 = data_i;
 86    case 14
 87        otherMakersContinues.data14 = data_i;
 88    case 15
 89        otherMakersContinues.data15 = data_i;
 90    case 16
 91        otherMakersContinues.data16 = data_i;
 92    case 17
 93        otherMakersContinues.data17 = data_i;
 94    case 18
 95        otherMakersContinues.data18 = data_i;
 96    case 19
 97        otherMakersContinues.data19 = data_i;
 98    case 20
 99        otherMakersContinues.data20 = data_i;
100    otherwise
101        otherMakersContinues.data1 = data_i;
