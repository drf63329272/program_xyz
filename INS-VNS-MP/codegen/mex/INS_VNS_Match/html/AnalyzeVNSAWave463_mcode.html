  2%% 实时 分析马克点的 连续 位置曲线、 速度曲线、 加速度曲线、 加速度曲线的波形
  3% 只计算一个时刻
  4
  5function [ otherMakersContinues,A_k_waves_OKLast ] = AnalyzeVNSAWave( otherMakers,k_vision,i_marker,otherMakersContinues,...
  6    parametersSet,visionFre,A_k_waves_OKLast )
  7
  8% persistent A_k_waves_OKLast
  9% if isempty(A_k_waves_OKLast)
 10%     A_k_waves_OKLast = zeros(3,1);  % 记录上一时刻判断成功的点（防止判断成功的点被覆盖）
 11% end
 12
 13waveThreshold_VNSAcc = parametersSet.waveThreshold_VNSAcc ;
 14INSVNSCalibSet = parametersSet.INSVNSCalibSet;
 15%% 更新连续线段 位置
 16
 17[~,ConPosition_i,ConVelocity_i,ConAcc_i,AWave] = Read_otherMakersContinues_i( otherMakersContinues,i_marker ); % 更新前的连续线段
 18dataN_i_PValid = otherMakersContinues.dataN( 1,i_marker );  % 第 i_marker 马克点 位置有效长度
 19
 20[ ConPosition_i,dataN_i_P_new] = GetContinuesPosition_2...
 21    ( otherMakers,k_vision,i_marker,ConPosition_i,dataN_i_PValid );
 22% 将新增加的点 NewPosition_i 更新到  otherMakersContinues
 23otherMakersContinues = Write_otherMakersContinues_i( otherMakersContinues,i_marker,ConPosition_i,1,dataN_i_P_new );
 24
 25if dataN_i_P_new>0
 26    %% 计算马克点速度
 27    [ Velocity_k,k_calV ] = CalVelocity( ConPosition_i,dataN_i_P_new,visionFre,INSVNSCalibSet.dT_CalV_Calib,INSVNSCalibSet.MinXYVNorm_CalAngle ) ;
 28    if k_calV>0  && ~isnan(Velocity_k(1))
 29        ConVelocity_i(:,k_calV) = Velocity_k ;
 30        
 31        otherMakersContinues = Write_otherMakersContinues_i( otherMakersContinues,i_marker,ConVelocity_i,2,k_calV );
 32        
 33         %% 计算马克点加速度
 34        [ acc_k,k_calA ] = CalVelocity( ConVelocity_i(1:3,:),k_calV,visionFre,INSVNSCalibSet.dT_CalV_Calib,INSVNSCalibSet.MinXYVNorm_CalAngle ) ;
 35       if k_calA>0  && ~isnan(acc_k(1))
 36 %% ？？？？？？？？ 不知为何反号了 ？？？？？？？？？？？？？？？？？？？？？？？？？
 37            acc_k = -acc_k;  
 38            ConAcc_i(:,k_calA) = acc_k ;
 39            otherMakersContinues = Write_otherMakersContinues_i( otherMakersContinues,i_marker,ConAcc_i,3,k_calA );
 40            
 41           %% ConAcc 波形分析
 42            ConAcc_i_valid = ConAcc_i(1:3,1:k_calA) ;
 43            A_WaveFlag  = AWave( (14:16)-13,:);  
 44            A_V = AWave((17:21)-13,:); 
 45            A_Acc_waveFront = AWave((22:24)-13,:); 
 46            A_Acc_waveBack = AWave((25:27)-13,:);   
 47              %% 启动搜索
 48            
 49             [ A_WaveFlag,A_V,A_Acc_waveFront,A_Acc_waveBack,A_k_waves_OKLast ] = AnalyzeWave...
 50                ( ConAcc_i_valid,k_calA,visionFre,A_V,A_WaveFlag,A_k_waves_OKLast,A_Acc_waveFront,A_Acc_waveBack,waveThreshold_VNSAcc );
 51            
 52            AWave = [  A_WaveFlag;A_V; A_Acc_waveFront; A_Acc_waveBack ];
 53            otherMakersContinues = Write_otherMakersContinues_i( otherMakersContinues,i_marker,AWave,4,A_k_waves_OKLast );
 54       end
 55           
 56    end
 57      
 58end
