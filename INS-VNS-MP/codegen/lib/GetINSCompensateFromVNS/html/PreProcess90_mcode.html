187%% 视觉位置预处理
188% 1）将视觉数据  从视觉世界坐标系 转到 北东地坐标系
189function otherMakers = PreProcess( otherMakers,BodyDirection  )
190coder.inline('never');
191global CalStartVN CalEndVN 
192%% 先转到北东地的同名坐标系
193Cv_r1 = [ 0 0 1; -1 0 0; 0 -1 0 ];
194% dbstop in BodyDirection2Cr_r1
195Cr_r1 = BodyDirection2Cr_r1( BodyDirection );   %   （法1）    要求人朝视觉标定标定的世界坐标系进行对准
196Cvr = Cr_r1' * Cv_r1 ;
197Cvr = Cv_r1 ;         %   （法2）    要求视觉世界坐标系的Z轴朝向正北
198
199%% 先转到 惯性系 的北东地，并以视觉的初始点为原点（不处理高度）
200
201Position_1 = otherMakers(1).Position(:,1) ;
202Position_1(2) = 0 ;      % 不处理高度方向
203
204for k=CalStartVN:CalEndVN
205    Position_k = otherMakers(k).Position ;
206    if ~isempty(Position_k)
207%         m = size(Position_k,2);
208%         position_offest = repmat(Position_1,1,m);     
209%         Position_k_new = Cvr*(Position_k-position_offest) ;  % 从视觉世界坐标系 转到 北东地坐标系
210        
211        Position_k_new = Cvr*Position_k;
212        otherMakers(k).Position = Position_k_new; 
213    else
214        otherMakers(k).Position = NaN(3,1); 
215    end
216    otherMakers(k).CalculatedTime = otherMakers(k).CalculatedTime+1 ;
217    if coder.target('MATLAB') && otherMakers(k).CalculatedTime~=1
218       fpritnf( 'otherMakers(%0.0f).CalculatedTime = %0.0f\n',k,otherMakers(k).CalculatedTime ); 
219    end
220end
