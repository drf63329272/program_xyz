535
536%% 马克点判定 3) 短时间（dT=3）运动的过程，惯性和视觉位移差 dPError_dT 进行判定， dPError_dT(i) = normest(dP_Inertial-dP_Vision)
537
538function [ trackedMakerPosition_k_OK,TrackFlag,min_dT_k,dPErrorNorm_dT_Min,dPError_dT_z_Min,dP_Inertial_xyNorm_Min,angleErr_dT_Min ]...
539    = Track_dT_Judge( otherMakers_k,InertialPosition,inertial_k,inertial_dT_k_last,trackedMakerPosition_last_k_dT,makerTrackThreshold )
540
541% global InertialData
542trackedMakerPosition_k_OK = NaN ;
543otherMakersPosition_k = otherMakers_k.Position ;
544TrackFlag = 0;
545
546M = size(otherMakersPosition_k,2) ;
547dP_Inertial = InertialPosition(:,inertial_k) - InertialPosition(:,inertial_dT_k_last);
548dP_Inertial_xy = dP_Inertial(1:2);
549dP_Inertial_z = dP_Inertial(3);  % 向下为正
550dPErrorNorm_dT  = zeros(1,M);
551dPError_dT_xy  = zeros(1,M);
552dPError_dT_z  = zeros(1,M);
553dP_Vision = zeros(3,M);
554angleErr_dT  = NaN(1,M);
555angleErr_dT_Min = NaN;
556%% 位移差 模， 角度差
557for i=1:M
558    otherMakersPosition_k_i = otherMakersPosition_k(:,i);
559    dP_Vision(:,i) = otherMakersPosition_k_i - trackedMakerPosition_last_k_dT;
560    % 位移差
561    dPErrorNorm_dT(i) = normest(dP_Inertial-dP_Vision(:,i)) ;
562    dPError_dT_xy(i) = normest(dP_Inertial_xy-dP_Vision(1:2,i)) ;
563    dPError_dT_z(i) = dP_Inertial_z-dP_Vision(3,i);  
564    % 低头时，dP_Inertial_z和dP_Vision(3,i)都为正，dP_Vision(3,i)的模更大，dPError_dT_z(i)为负
565    
566    % 角度差
567    temp = dP_Inertial'*dP_Vision(:,i) / normest(dP_Inertial) / normest(dP_Vision(:,i)) ;
568    angleErr_dT(i) = acos(temp) ;
569end
570
571
572
573[dPErrorNorm_dT_Min,min_dT_k] = min(dPErrorNorm_dT,[],2);  % 取位移差最小的点判断
574dPError_dT_z_Min = dPError_dT_z(min_dT_k);
575% 惯性位移大小 ： 评价静止还是运动
576otherMakersPosition_k_min = otherMakersPosition_k(:,min_dT_k);
577dP_Inertial_Norm = normest(dP_Inertial) ;  
578dP_Inertial_xyNorm_Min = normest(dP_Inertial(1:2)) ;
579dP_Inertial_z_Min = dP_Inertial(3) ;
580if dP_Inertial_Norm < makerTrackThreshold.MaxStaticDisp_dT  
581    %% 准静止状态，用dS位移判断
582    if dPErrorNorm_dT_Min > makerTrackThreshold.PositionErrorBear_dT
583        % 马克点明显运动，剔除
584        TrackFlag = -3.1 ;
585    else
586        % 惯性视觉运动位移差小，且连续于跟踪OK的马克点，则跟踪OK。
587        %% 连续判断出错时会误判
588        if otherMakers_k.ContinuesFlag(min_dT_k)==1
589            trackedMakerPosition_k_OK = otherMakersPosition_k(:,min_dT_k) ;
590            TrackFlag = 3.1;
591            return;
592        else
593            % 惯性视觉运动位移差小，但不连续，进入4）判断
594            TrackFlag = 1.31;
595            return;
596        end
597    end
598    
599else
600    %% 运动状态：
601    if otherMakers_k.ContinuesFlag(min_dT_k)==1
602       %% 与前一时刻跟踪成功的点连续则放宽要求
603       PositionErrorBear_dT = makerTrackThreshold.PositionErrorBear_dT*makerTrackThreshold.ContinuesTrackedMagnifyRate ;
604       Max_dPAngle_dS=  makerTrackThreshold.Max_dPAngle_dS*makerTrackThreshold.ContinuesTrackedMagnifyRate ;
605       MaxHighMoveErrRate = makerTrackThreshold.MaxHighMoveErrRate*makerTrackThreshold.ContinuesTrackedMagnifyRate ;
606    else
607        PositionErrorBear_dT = makerTrackThreshold.PositionErrorBear_dT ;
608       Max_dPAngle_dS=  makerTrackThreshold.Max_dPAngle_dS ;
609       MaxHighMoveErrRate = makerTrackThreshold.MaxHighMoveErrRate ;
610    end
611    % 静止点直接剔除
612    if normest( dP_Vision(:,min_dT_k) )<makerTrackThreshold.MaxStaticDisp_dT
613        TrackFlag=1.35;
614        return;
615    end
616    %% 高度变化大，水平方向不可信，直接通过高度判定
617    if normest(dP_Inertial_z_Min) > makerTrackThreshold.BigHighMove
618       HighMoveErr = dPError_dT_z(min_dT_k) / dP_Inertial_z_Min ;
619       if HighMoveErr>MaxHighMoveErrRate(1) && HighMoveErr < MaxHighMoveErrRate(2)
620           %% 高度方向位移误差比例小，跟踪OK
621           TrackFlag = 3.9 ;
622           trackedMakerPosition_k_OK = otherMakersPosition_k(:,min_dT_k) ;
623           return;
624       end
625    end
626    %% （角度差判断为主）当运动距离超过 moveDistance 时，判断角度差，并放宽位移差条件
627    if dP_Inertial_Norm > makerTrackThreshold.moveDistance
628       if angleErr_dT(min_dT_k) <  makerTrackThreshold.Max_dPAngle_dS && dPErrorNorm_dT_Min < makerTrackThreshold.MaxPositionError_dS
629            trackedMakerPosition_k_OK = otherMakersPosition_k_min ;
630            TrackFlag = 3.7;
631            angleErr_dT_Min = angleErr_dT(min_dT_k) ;
632            return;
633       end
634    end
635    %% 位移差判断为主
636    if dPErrorNorm_dT_Min < PositionErrorBear_dT  && angleErr_dT(min_dT_k) <  Max_dPAngle_dS*2
637        trackedMakerPosition_k_OK = otherMakersPosition_k_min ;
638        TrackFlag = 3.5;
639    %     fprintf( '3.5）惯性视觉位移差长度=%0.4f ，跟踪OK\n',normest(dPError_dT) );
640        return;
641    end
642    TrackFlag=1.37;
643end
644
645if TrackFlag==0
646   disp('error')  
647end

649%% 马克点连续性判断
650% dPi_ConJudge ： 连续性判断指标的大小：前后两帧的位移模
651% ContinuesFlag = 0   不连续
652%               =1    连续，且是与跟踪成功马克点连续
653%               =2   连续，和跟踪失败的点连续
