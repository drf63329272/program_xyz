223%% 根据 BodyDirection 求姿态
224% 符号有待验证！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！
225%   BodyDirection  人朝北为 [1 0 0]
226function Cr_r1 = BodyDirection2Cr_r1( BodyDirection )
227coder.inline('never');
228
229V = [1 0 0];   % 正北
230thita = acos( V*BodyDirection / normest(BodyDirection)   );
231% 通过叉乘可判断角度方向
232temp = cross(V,BodyDirection) ;  % 从 V 到 BodyDirection 逆时针180°以内时，temp朝上，temp(3)<0
233if temp(3)>0
234        % 从 V 到 BodyDirection 逆时针转动超过180°
235      thita = -thita ;
236end
237    
238Cr_r1 = RotateZ( thita ) ;
