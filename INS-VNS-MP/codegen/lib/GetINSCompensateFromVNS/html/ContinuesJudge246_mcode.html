650%% 马克点连续性判断
651% dPi_ConJudge ： 连续性判断指标的大小：前后两帧的位移模
652% ContinuesFlag = 0   不连续
653%               =1    连续，且是与跟踪成功马克点连续
654%               =2   连续，和跟踪失败的点连续
655function [ otherMakers_k,dPi_ConJudge ] = ContinuesJudge( otherMakers_k,otherMakers_k_last,trackedMakerPosition,k_vision,makerTrackThreshold )
656
657
658% 前一时刻跟踪成功时，给出当前点是否相对跟踪成功的点连续。  Continues = 1
659% 前一时刻跟踪失败时，给出当前每个点是否为连续点的结果,且记录着个点能连续到往前最早（但不超过dT）的点的位置和时间。
660%        Continues = 2 ，
661
662% global inertialFre visionFre  moveDistance
663% 记录每个马克点的连续特性
664MaxOtherMarkerBufN = size(otherMakers_k.ContinuesFlag,2);
665M = otherMakers_k.otherMakersN ;
666otherMakers_k.ContinuesFlag = zeros(1,MaxOtherMarkerBufN) ; % 不连续
667otherMakers_k.ContinuesLastPosition = NaN(3,MaxOtherMarkerBufN) ;
668otherMakers_k.ContinuesLastTime = NaN(1,MaxOtherMarkerBufN) ;
669otherMakersPosition_k = otherMakers_k.Position ;  
670
671dPi_ConJudge=nan;
672
673if k_vision>1 
674    
675    if ~isnan(trackedMakerPosition(1,k_vision-1))
676        %% 只判断当前马克点是否与前时刻跟踪成功的马克点连续
677        trackedMakerPosition_kLast = trackedMakerPosition(:,k_vision-1) ;
678        otherMakersPosition_k_Dis = otherMakersPosition_k(:,1:M)-repmat(trackedMakerPosition_kLast,1,M) ;
679        otherMakersPosition_k_Dis_Norm = zeros(1,M);
680        for j=1:M
681            otherMakersPosition_k_Dis_Norm(j) = normest(otherMakersPosition_k_Dis(:,j));
682        end
683        [dPi_ConJudge,minCon_k] = min(otherMakersPosition_k_Dis_Norm);  
684        if dPi_ConJudge < makerTrackThreshold.MaxContinuesDisplacement
685    %         trackedMakerPosition_k_OK = otherMakersPosition_k(:,m) ;
686    %         TrackFlag = 1;
687    %         fprintf('马克点连续：位移=%0.4f，跟踪OK \n',Min_otherMakersPosition_k_Dis_Norm);
688            
689            otherMakers_k.ContinuesFlag(minCon_k) = 1 ; % 连续，且是与跟踪成功马克点连续
690            otherMakers_k.ContinuesLastPosition(:,minCon_k) = trackedMakerPosition_kLast ;
691            otherMakers_k.ContinuesLastTime(minCon_k) = otherMakers_k_last.time ;
692            otherMakers_k.ContinuesLastK(minCon_k) = k_vision-1 ;
693        end
694        
695    else
696        %% 判断当前马克点是否为连续马克点，记录每个点对应的最早（但不超过dT）连续点
697        M_last = otherMakers_k_last.otherMakersN ;
698        if M_last==0
699            % 上时刻无马克点
700            for i=1:M
701                otherMakers_k.ContinuesFlag(i) = 0 ; % 不连续
702                otherMakers_k.ContinuesLastPosition(:,i) = NaN ;
703                otherMakers_k.ContinuesLastTime(i) = NaN ;
704                otherMakers_k.ContinuesLastK(i) = NaN ;
705            end
706            dPi_ConJudge=nan;
707            return;
708        end
709        % 一共有 M*M_last 种组合
710        for i=1:M
711            dPi = repmat(otherMakers_k.Position( :,i ),1,M_last)- otherMakers_k_last.Position ;
712            dPiNorm = zeros(1,M_last);
713            for j=1:M_last
714                dPiNorm(j) = normest(dPi(:,j));
715            end
716            
717            [dPi_ConJudge,min_i] = min(dPiNorm);   
718            if normest(dPi_ConJudge) < makerTrackThreshold.MaxContinuesDisplacement
719                %  otherMakers_k.Position( :,i ) 与 otherMakers_k_last.Position(:,min_i) 连续
720                % 找到一个连续的点，记录上一点
721                otherMakers_k.ContinuesFlag(i) = 2 ; % 连续，和跟踪失败的点连续
722                % 如果前一个点为连续点，则将前一个点的连续记录传递过来
723                if otherMakers_k_last.ContinuesFlag(min_i) == 2
724                    otherMakers_k.ContinuesLastK(i) = otherMakers_k_last.ContinuesLastK(min_i) ; % 传递记录上一个时刻存储的连续信息
725                    otherMakers_k.ContinuesLastPosition(:,i) = otherMakers_k_last.ContinuesLastPosition(:,min_i) ;
726                    otherMakers_k.ContinuesLastTime(i) = otherMakers_k_last.ContinuesLastTime(min_i);
727                elseif otherMakers_k_last.ContinuesFlag(min_i) == 0
728                    otherMakers_k.ContinuesLastK(i) = k_vision-1 ; % 直接记录上一个时刻
729                    otherMakers_k.ContinuesLastPosition(:,i) = otherMakers_k_last.Position( :,min_i ) ;
730                    otherMakers_k.ContinuesLastTime(i) = otherMakers_k_last.time ;                    
731                elseif otherMakers_k_last.ContinuesFlag(min_i) == 1 
732                    % 与跟踪成功点连续检测成功，但跟踪识别失败的情况，传递到现在。如果传递时间超过2秒，则不再传递。
733                    if (otherMakers_k_last.ContinuesLastTime(min_i)-otherMakers_k.time) > 20
734                        otherMakers_k.ContinuesFlag(i) = 2 ;
735                    else
736                        otherMakers_k.ContinuesFlag(i) = 1 ;
737                    end                    
738                    otherMakers_k.ContinuesLastK(i) = otherMakers_k_last.ContinuesLastK(min_i) ; % 传递记录上一个时刻存储的连续信息
739                    otherMakers_k.ContinuesLastPosition(:,i) = otherMakers_k_last.ContinuesLastPosition(:,min_i) ;
740                    otherMakers_k.ContinuesLastTime(i) = otherMakers_k_last.ContinuesLastTime(min_i);
741                end
742            else
743                otherMakers_k.ContinuesFlag(i) = 0 ; % 不连续
744                otherMakers_k.ContinuesLastPosition(:,i) = NaN ;
745                otherMakers_k.ContinuesLastTime(i) = NaN ;
746                otherMakers_k.ContinuesLastK(i) = NaN ;
747            end
748                        
749        end
750        
751    end
752else
753    dPi_ConJudge=nan;
754end

759% 且该点trackedMakerPosition跟踪成功
