166%% 判断从 search_k_start 到 search_k_end 的这一段速度是否满足条件
167% 1） 速度z模小于 MaxVZ_Calib
168% 2） 速度xy模大于 MinVXY_Calib
169function IsCalibDataVelocityOK = CalibDataVelocityJudge( trackedMarkerVelocity_k,INSVNSCalibSet )
170%% 阈值参数设置
171
172MaxVZ_Calib = INSVNSCalibSet.MaxVZ_Calib;     % m/s Z方向速度最大绝对值
173MinVXY_Calib = INSVNSCalibSet.MinVXY_Calib ;    % m/s XY 平面速度模最小绝对值
174
175IsCalibDataVelocityOK = 0 ;
176if isnan(trackedMarkerVelocity_k(1))
177   return; 
178end
179if abs(trackedMarkerVelocity_k(3)) > MaxVZ_Calib
180    return;
181end
182if trackedMarkerVelocity_k(4) < MinVXY_Calib
183    return;    
184end
185IsCalibDataVelocityOK = 1 ;
