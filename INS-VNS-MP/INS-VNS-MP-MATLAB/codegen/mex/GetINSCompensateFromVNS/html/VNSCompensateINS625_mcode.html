112%% 位移补偿
113%%% Input
114% trackedMakerPosition_InertialTime ： 马克点光学位置，按惯性时序存储
115% HipDisplacement ： Hip在北东地下的位置 （由 BVH 得到）
116% InertialPosition： 惯性系与马克点安装位置对应关节的位置（安装在头上时，InertialPosition 为惯性头的位置）
117%%% Output
118% InertialPositionNew ： 补偿后惯性的位置
119% HipDisplacementNew： 补偿后Hip的位置
120function [ InertialPositionCompensate_out,HipDisplacementNew_out ] = VNSCompensateINS...
121( compensateRate,trackedMakerPosition_InertialTime,HipDisplacement,InertialPosition )
122coder.inline('never');
123coder.extrinsic('fprintf');
124coder.extrinsic('DrawCompensate');
125global CalStartIN  CalEndIN IsOnLine
126
127%% BVH 读取的个数 N_BVH 可能会比 N1 多几个
128N_BVH = size(HipDisplacement,2);
129N1 = size(trackedMakerPosition_InertialTime,2);
130for k=N1+1:N_BVH
131    trackedMakerPosition_InertialTime(:,k) = trackedMakerPosition_InertialTime(:,N1);
132    InertialPosition(:,k) = InertialPosition(:,N1);
133end
134
135%% 先补偿 InertialPositionNew
136persistent InertialPositionCompensate  InertialPositionNew HipDisplacementNew
137if isempty(InertialPositionCompensate)
138    InertialPositionCompensate = zeros(3,N_BVH); % 每一步的累积位移补偿量 记录
139    InertialPositionNew = NaN(3,N_BVH);
140    InertialPositionNew(3,:) = InertialPosition(3,:); % 高度不补偿
141    InertialPositionNew(1:2,1) = InertialPosition(1:2,1) ;  % 起始点选择惯性
142    
143    HipDisplacementNew = HipDisplacement ;
144end
145persistent  InertialErr StepCompensate
146if isempty(StepCompensate)
147    InertialErr = zeros(2,N_BVH);       % 实时记录补偿后的误差
148    StepCompensate = zeros(2,N_BVH);  % 单步补偿量
149end
150
151for k = max(2,CalStartIN) : CalEndIN 
152    % 先用纯惯性递推
153    InertialPositionNew(1:2,k) = InertialPositionNew(1:2,k-1)+( InertialPosition(1:2,k)-InertialPosition(1:2,k-1) );
154    % 先求纯惯性为误差
155    if ~isnan(trackedMakerPosition_InertialTime(1,k))
156        InertialErr(:,k) = trackedMakerPosition_InertialTime(1:2,k) - InertialPositionNew(1:2,k) ;
157        % 补偿误差
158        InertialPositionNew(1:2,k) = InertialPositionNew(1:2,k) + InertialErr(:,k)*compensateRate ; 
159        StepCompensate(:,k) = InertialErr(:,k)*compensateRate ;
160    end
161    InertialPositionCompensate(1:2,k) = InertialPositionNew(1:2,k) - InertialPosition(1:2,k) ;  % 累积位移补偿量
162end
163
164%% 通过 InertialPositionNew 计算 HipDisplacementNew
165% 将Head位置传到Hip，：保持头和head的相对位移
166
167for k=CalStartIN:CalEndIN    
168    HipDisplacementNew( 1:2,k ) = HipDisplacement(1:2,k) + InertialPositionCompensate(1:2,k) ;
169end
170
171InertialPositionCompensate_out = InertialPositionCompensate ;
172HipDisplacementNew_out = HipDisplacementNew;
173if IsOnLine==0
174    DrawCompensate( InertialErr,StepCompensate,trackedMakerPosition_InertialTime,...
175        InertialPosition,InertialPositionNew,InertialPositionCompensate,compensateRate,N_BVH ) ;
176end
