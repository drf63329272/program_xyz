 3%% 先进先出形式 增加数据
 4
 5% 不允许扩容，超出则将前面的数据挤出去
 6% data [M*N]
 7% validN ： data有效数据个数 data(:,1:validN) 有效
 8% dataAdd ： 新增数据
 9
10function [ data,validN,removeN ] = AddListData( data,validN,dataAdd )
11
12[M,dataN] = size(data);
13dataAddN = size(dataAdd,2);
14if validN > dataN
15   fprintf('error (AddListData):validN > dataAddN '); 
16end
17if size(dataAdd,1) ~= M
18   fprintf('error (AddListData):validN > size(dataAdd,1) ~= size(data,1) ');  
19end
20
21removeN = (validN+dataAddN)-dataN ;  % 需要从前面挤出去数据个数
22if removeN > 0
23   % 需要挤掉前面 removeN 个数据
24   
25   data_BackValid_N = dataN-dataAddN ;
26   for i=1:data_BackValid_N
27       data(:,i) = data( :, removeN+i );  % 保留后面有效部分
28   end
29%    data(:,1:data_BackValid_N) = data( :, removeN+1 : validN );  % 保留后面有效部分
30   data(:,data_BackValid_N+1:dataN) = dataAdd;
31   
32   
33   validN = dataN;
34else
35    % 空闲足够
36    data( :,validN+1:validN+dataAddN ) = dataAdd;
37    validN = validN+dataAddN;
38end
