1%% 利用 otherMakers(k).ContinuesLasti 将上时刻的连续线段进行重新排序
 2%% 对象： 速度 加速度 波形参数 等
 3% ContinuesLasti 当前时刻 所有马克点 对应的 上时刻马克点序号
 4function [ otherMakersContinues_New,A_k_waves_OKLast_All_New ] = ReOrderContinues( ContinuesLasti,otherMakersContinues,A_k_waves_OKLast_All ) 
 5%% 不需要排序
 6otherMakersN_Last = otherMakersContinues.otherMakersN;
 7otherMakersN_new = length(ContinuesLasti);
 8
 9if   otherMakersN_Last==otherMakersN_new
10    if otherMakersN_Last == 0 
11        otherMakersContinues_New = otherMakersContinues; % 不需要排序
12        A_k_waves_OKLast_All_New = A_k_waves_OKLast_All;
13        return;
14    end
15
16    err = sum(ContinuesLasti - (1:otherMakersN_Last));
17    if err == 0
18        otherMakersContinues_New = otherMakersContinues; % 不需要排序
19        A_k_waves_OKLast_All_New = A_k_waves_OKLast_All;
20        return;
21    end
22end
23
24%% 需要排序
25visualN = size(otherMakersContinues.data1,2);
26otherMakersContinues_New = Initial_otherMakersContinues( visualN );
27
28otherMakersContinues_New.otherMakersN = otherMakersN_new;
29A_k_waves_OKLast_All_New = zeros(3,20);
30
31for i=1:otherMakersN_new
32    if ~isnan(ContinuesLasti(i)) && ContinuesLasti(i)>0    % otherMakersContinues 有值     
33        otherMakersContinues_New.dataN(:,i) = otherMakersContinues.dataN(:,ContinuesLasti(i));
34        
35        data_i = Read_otherMakersContinues_i( otherMakersContinues,ContinuesLasti(i) );
36        otherMakersContinues_New = Write_otherMakersContinues_i( otherMakersContinues_New,i,data_i,0,Inf ); % 最后一个参数无效
37        
38        A_k_waves_OKLast_All_New(:,i) = A_k_waves_OKLast_All(:,ContinuesLasti(i));
39    else
40        otherMakersContinues_New.dataN(i) = 0;
41        A_k_waves_OKLast_All_New(:,i) = zeros(3,1);
42    end
43end
