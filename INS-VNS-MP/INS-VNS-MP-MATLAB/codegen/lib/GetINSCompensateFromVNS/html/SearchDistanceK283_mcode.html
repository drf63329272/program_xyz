759%% 查找惯性k时刻前运动了大于 dS 距离的最近的点
760% 且该点trackedMakerPosition跟踪成功
761function dS_Inertial_last_k = SearchDistanceK( InertialPosition,kCurrent,dS,trackedMakerPosition )
762global inertialFre InertialData_visual_k
763stepT = 1 ;  % 搜索步长时间
764MaxSearchT = 60 ; % 最长搜索时间长度
765stepK = fix(inertialFre*stepT) ; % 搜索步长
766
767maxSearchK = min( kCurrent-1,fix(inertialFre*MaxSearchT) );
768InertialPosition_kSea = InertialPosition(:,kCurrent);
769dS_Inertial_last_k = NaN ;     % 默认设置：寻找失败
770%wh = waitbar(0,'SearchDistanceK');
771for i=1:stepK:maxSearchK
772    dP = InertialPosition(:,kCurrent-i) - InertialPosition_kSea ;
773    distance = normest( dP ); 
774    if distance > dS
775%         kCurrent_Vision = InertialK_to_VisionK(kCurrent-i);
776        kCurrent_Vision = InertialData_visual_k(kCurrent-i);
777        if ~isnan(trackedMakerPosition(1,kCurrent_Vision))            
778            % 距离满足，且 trackedMakerPosition 跟踪成功
779            dS_Inertial_last_k = double(kCurrent-i) ;
780            break;
781        end
782    end
783 %   waitbar(i/kCurrent);
784end
785%close(wh)
786disp('')
